void taskCreateHook(int *p) {
	p-=16;
	if (p[0]==(int)0xFF821ACC)  p[0]=(int)mykbd_task;
	if (p[0]==(int)0xFF842A44)  p[0]=(int)jogdial_task_my; 
	if (p[0]==(int)0xFF85A004)  p[0]=(int)movie_record_task;
	if (p[0]==(int)0xFF85E03C)  p[0]=(int)capt_seq_task;
	if (p[0]==(int)0xFF878F84)  p[0]=(int)init_file_modules_task;
	if (p[0]==(int)0xFF8B6864)  p[0]=(int)exp_drv_task;
	if (p[0]==(int)0xFFA10248)  p[0]=(int)filewritetask;
}

void boot()
{
	long *canon_data_src = (void*)0xFFB1E480;
	long *canon_data_dst = (void*)0x1900;
	long canon_data_len = 0xF164 - 0x1900; // data_end - data_start
	long *canon_bss_start = (void*)0xF164; // just after data
	long canon_bss_len = 0xCBD40 - 0xF164;

    long i;


    // Enable CPU caches and MPU
    asm volatile (
    "MRC     p15, 0, R0,c1,c0\n"
    "ORR     R0, R0, #0x1000\n"
    "ORR     R0, R0, #4\n"
    "ORR     R0, R0, #1\n"
    "MCR     p15, 0, R0,c1,c0\n"
    :::"r0");

    for(i=0;i<canon_data_len/4;i++)
    canon_data_dst[i]=canon_data_src[i];

    for(i=0;i<canon_bss_len/4;i++)
    canon_bss_start[i]=0;

    // jump to init-sequence that follows the data-copy-routine
    asm volatile ("B      sub_FF8101A4_my\n");
};
